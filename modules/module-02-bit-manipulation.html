Brian Kernighan's Algorithm — Count Set Bits

#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter a number: ";
    cin >> n;
    
    int original_n = n;
    int count = 0;
    
    // We will keep looping as long as n is not 0
    while (n) {
        // This is Brian Kernighan's trick!

        // Subtracting 1 from n flips all the bits after the rightmost '1', including the rightmost '1' itself.
        // For example, if n = 12 (1100 in binary), then n-1 = 11 (1011 in binary).
        // Doing a bitwise AND between n and (n-1) will turn off that rightmost '1'.
        // 1100 & 1011 = 1000 (We just removed the rightmost 1!)
        n &= (n - 1); 
        
        // Since we successfully removed a '1', we count it
        count++;
    }
    
    cout << "Number of set bits in " << original_n << " is: " << count << endl;
    return 0;
}
// Note: In competitive programming, C++ has a super fast built-in function for this: __builtin_popcount(n)
if __name__ == "__main__":
    n = int(input("Enter a number: "))
    original_n = n
    count = 0
    
    # Python integers have infinite precision.
    # We mask to 32 bits to handle negative numbers properly.
    n = n & 0xFFFFFFFF
    
    # We will keep looping as long as n is not 0
    while n:
        # This is Brian Kernighan's trick!
        # n & (n - 1) essentially drops the lowest set bit (the rightmost '1').
        # This is much faster than checking every single bit one by one!
        n &= (n - 1)  
        
        # Since we successfully removed a '1', we count it
        count += 1
        
    print(f"Number of set bits in {original_n} is: {count}")

# Note: Python makes this really easy built-in:
# bin(original_n).count('1') turns the number into a binary string and counts the '1's!
