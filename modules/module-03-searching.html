📖 Theory

📋 Linear Search

Linear Search checks every element sequentially until the target is found or the array ends.

When to use: Unsorted arrays, small datasets, or when you need to find all occurrences.

Time: O(n) — Must check each element
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, target;
    cout << "Enter number of elements: ";
    cin >> n;
    
    vector<int> arr(n);
    cout << "Enter " << n << " elements: ";
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    cout << "Enter target to search: ";
    cin >> target;
    
    int index = -1;
    // Loop through every single element in the array from start to finish
    for (int i = 0; i < n; i++) {
        // If the current element matches what we are looking for
        if (arr[i] == target) {
            index = i; // Success! Save the position
            break;     // Stop searching
        }
    }
    
    if (index != -1) {
        cout << "Element found at index " << index << endl;
    } else {
        cout << "Element not found" << endl;
    }
    
    return 0;
}
if __name__ == "__main__":
    n = int(input("Enter number of elements: "))
    
    print(f"Enter {n} elements separated by space: ")
    arr = list(map(int, input().split()))
    
    target = int(input("Enter target to search: "))
    
    index = -1
    # enumerate gives us both the index (i) and the value (val)
    for i, val in enumerate(arr):
        # If the current value matches our target
        if val == target:
            index = i  # Success! Save the position (index)
            break
            
    if index != -1:
        print(f"Element found at index {index}")
    else:
        print("Element not found")

⚡ Binary Search

Binary Search works on sorted arrays. Compare the target with the middle element; if not equal, eliminate half the array and repeat.

Key Requirement: Array must be sorted!

Pattern:

  1. Set low = 0, high = n-1
  2. While low ≤ high: compute mid = low + (high - low) / 2
  3. If arr[mid] == target: found!
  4. If arr[mid] < target: search right half (low = mid + 1)
  5. If arr[mid] > target: search left half (high = mid - 1)
Time: O(log n) — Halving search space each step
⚠️ Avoid Overflow
Use mid = low + (high - low) / 2 instead of (low + high) / 2 to prevent integer overflow!
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, target;
    cout << "Enter number of elements (must be sorted): ";
    cin >> n;
    
    vector<int> arr(n);
    cout << "Enter " << n << " sorted elements: ";
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    cout << "Enter target to search: ";
    cin >> target;
    
    int index = -1;
    // low and high act as our search boundaries
    int low = 0, high = n - 1;
    
    // Keep searching as long as our boundaries haven't crossed
    while (low <= high) {
        // Find the middle element.
        // We use low + (high - low) / 2 instead of (low + high) / 2
        // to prevent integer overflow if low and high are very large!
        int mid = low + (high - low) / 2;
        
        if (arr[mid] == target) {
            index = mid; // Jackpot! We found the target
            break;
        }
        else if (arr[mid] < target) {
            // Target is strictly greater than the middle element.
            // Since the array is sorted, we can ignore the entire left half!
            low = mid + 1;
        }
        else {
            // Target is strictly smaller than the middle element.
            // We can ignore the entire right half!
            high = mid - 1;
        }
    }
    
    if (index != -1) {
        cout << "Element found at index " << index << endl;
    } else {
        cout << "Element not found" << endl;
    }
    
    return 0;
}
if __name__ == "__main__":
    n = int(input("Enter number of elements (must be sorted): "))
    
    print(f"Enter {n} sorted elements separated by space: ")
    arr = list(map(int, input().split()))
    
    target = int(input("Enter target to search: "))
    
    index = -1
    # low and high act as our search boundaries
    low, high = 0, len(arr) - 1
    
    # Keep searching as long as our boundaries haven't crossed
    while low <= high:
        # Find the middle index. 
        # // is integer division in Python.
        mid = low + (high - low) // 2
        
        if arr[mid] == target:
            index = mid  # Jackpot! We found the target
            break
        elif arr[mid] < target:
            # Target is larger. The left half is useless now.
            low = mid + 1
        else:
            # Target is smaller. The right half is useless now.
            high = mid - 1
            
    if index != -1:
        print(f"Element found at index {index}")
    else:
        print("Element not found")

Suggested Teaching Sequence
  1. Begin with the theory section — walk through key concepts (~20 min)
  2. Use the interactive visualization to demonstrate the algorithm (~15 min)
  3. Solve the first problem together as a class (~20 min)

Common Student Mistakes

  • Confusing time and space complexity of different approaches
  • Jumping to code before understanding the algorithm logic
Discussion Questions
  • What is the brute force approach and why is it insufficient?
  • How does the optimal approach improve on the brute force?
  • Can you identify the pattern that connects this module's problems?
